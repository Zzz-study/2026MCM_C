def extract_features_from_data(df):
    """
    从题目数据集提取模型所需特征
    """
    features = {}
    
    for _, row in df.iterrows():
        player_id = f"{row['celebrity_name']}_{row['season']}"
        
        # 1. 技术基因评分（使用本周评委评分）
        judge_scores = [row[f'week{week}_judge{j}_score'] 
                       for week in range(1, 12) 
                       for j in range(1, 5) 
                       if pd.notna(row.get(f'week{week}_judge{j}_score'))]
        
        if judge_scores:
            tech_score = calculate_technical_gene(judge_scores[-4:])  # 取最近一周
        else:
            tech_score = 0.5  # 默认值
        
        # 2. 环境适应性（使用前三问估计的粉丝投票）
        fan_votes = estimate_fan_votes(row)  # 调用前三问模型
        adaptation = normalize_fan_votes(fan_votes)
        
        # 3. 创新性
        novelty = calculate_novelty(judge_scores[-4:]) if len(judge_scores) >= 4 else 0
        
        # 4. 经验（是否全明星）
        experience = 1 if row['season'] == 15 else 0
        
        # 5. 社交邻域（基于行业和舞伴）
        neighbors = find_social_neighbors(
            row['celebrity_industry'],
            row['ballroom_partner'],
            df
        )
        
        features[player_id] = {
            'tech_score': tech_score,
            'adaptation': adaptation,
            'novelty': novelty,
            'experience': experience,
            'neighbors': neighbors,
            'industry': row['celebrity_industry'],
            'partner': row['ballroom_partner']
        }
    
    return features

class FairnessMetrics:
    @staticmethod
    def controversy_index(fan_ranks, judge_ranks):
        """争议指数 = 粉丝排名与评委排名差异的平均值"""
        return np.mean(np.abs(fan_ranks - judge_ranks))
    
    @staticmethod
    def balance_index(top_tech_survival_rate, top_fan_survival_rate):
        """平衡指数 = 1 - |技术前20%晋级率 - 粉丝前20%晋级率|"""
        return 1 - abs(top_tech_survival_rate - top_fan_survival_rate)
    
    @staticmethod
    def meritocracy_score(eliminated_tech_rank, total_players):
        """精英保护分数 = 1 - (被淘汰选手中技术排名前30%的比例)"""
        tech_top_30 = eliminated_tech_rank <= 0.3 * total_players
        return 1 - (np.sum(tech_top_30) / len(eliminated_tech_rank))

class EntertainmentMetrics:
    @staticmethod
    def suspense_index(survival_scores):
        """悬念指数 = 淘汰候选池选手生存指数的变异系数"""
        if len(survival_scores) < 2:
            return 0
        return np.std(survival_scores) / np.mean(survival_scores)
    
    @staticmethod
    def surprise_index(eliminated_rank, candidate_ranks):
        """意外指数 = 淘汰选手在候选池中的排名（1为最可能被淘汰）"""
        if eliminated_rank not in candidate_ranks:
            return 0
        return 1 - (candidate_ranks.index(eliminated_rank) / len(candidate_ranks))
    
    @staticmethod
    def engagement_index(fan_vote_volatility):
        """参与度指数 = 粉丝投票波动率（反映观众反应强度）"""
        return np.std(fan_vote_volatility) / np.mean(fan_vote_volatility)

def historical_backtest(model, seasons_data, controversial_cases):
    """
    使用1-34季数据回测，验证模型效果
    """
    results = {
        'fairness_improvement': {},
        'entertainment_gain': {},
        'controversy_resolution': {}
    }
    
    for season_id, season_data in seasons_data.items():
        # 模拟新模型
        new_results = model.simulate_season(season_data)
        
        # 获取历史实际结果
        actual_results = season_data['actual_results']
        
        # 计算公平性指标
        new_fairness = calculate_fairness_metrics(new_results)
        old_fairness = calculate_fairness_metrics(actual_results, method='rank')  # 排名法
        percent_fairness = calculate_fairness_metrics(actual_results, method='percent')  # 百分比法
        
        # 计算娱乐性指标
        new_entertainment = calculate_entertainment_metrics(new_results)
        old_entertainment = calculate_entertainment_metrics(actual_results)
        
        # 记录改进
        results['fairness_improvement'][season_id] = {
            'vs_rank': new_fairness['balance'] - old_fairness['balance'],
            'vs_percent': new_fairness['balance'] - percent_fairness['balance']
        }
        
        results['entertainment_gain'][season_id] = {
            'suspense': new_entertainment['suspense'] - old_entertainment['suspense'],
            'surprise': new_entertainment['surprise'] - old_entertainment['surprise']
        }
        
        # 检查争议案例是否被修正
        if season_id in controversial_cases:
            case = controversial_cases[season_id]
            resolved = model_resolves_controversy(new_results, case)
            results['controversy_resolution'][season_id] = resolved
    
    return results

class SimplifiedDanceEcosystem:
    """
    简化版模型，只有2个核心调节参数
    """
    def __init__(self):
        # 固定参数（通过历史数据校准）
        self.fixed_params = {
            'D': 0.5,      # 扩散系数
            'λ': 0.3,      # 创新奖励
            'θ_novelty': 0.7,  # 创新阈值
            'mutation_rate': 0.1  # 随机事件概率
        }
        
        # 核心调节滑块（制片人可调节）
        self.sliders = {
            'professional_entertainment_balance': 0.5,  # 0=纯专业, 1=纯娱乐
            'suspense_intensity': 0.5,  # 0=低悬念, 1=高悬念
        }
    
    def calculate_weights(self, week, total_weeks):
        """
        根据平衡滑块计算动态权重
        """
        balance = self.sliders['professional_entertainment_balance']
        
        # 基础权重
        if week <= total_weeks / 3:
            base_tech, base_fan = 0.6, 0.3  # 早期重专业
        elif week <= 2 * total_weeks / 3:
            base_tech, base_fan = 0.45, 0.45  # 中期平衡
        else:
            base_tech, base_fan = 0.3, 0.6  # 后期重娱乐
        
        # 根据滑块调整
        tech_weight = base_tech + (1 - balance) * 0.2  # 滑块向左增加专业权重
        fan_weight = base_fan + balance * 0.2  # 滑块向右增加娱乐权重
        
        # 归一化
        total = tech_weight + fan_weight + 0.1  # 趋势分固定0.1
        return {
            'tech': tech_weight / total,
            'fan': fan_weight / total,
            'trend': 0.1 / total
        }
    
    def calculate_selection_pressure(self):
        """
        根据悬念强度滑块计算选择压力
        """
        suspense = self.sliders['suspense_intensity']
        # 悬念强度高 → 选择压力小 → 淘汰更随机
        # 悬念强度低 → 选择压力大 → 淘汰更确定
        return 1.2 - suspense * 0.7  # 范围[0.5, 1.2]

class ControlledRandomEvents:
    """
    可控的随机事件系统
    确保不确定性在制片人可控范围内
    """
    def __init__(self, enabled=True, max_impact=0.1):
        self.enabled = enabled
        self.max_impact = max_impact  # 最大影响幅度±10%
        self.events_log = []
    
    def trigger_event(self, week, players, survival_indices):
        """
        触发随机事件，确保可控性
        """
        if not self.enabled or random.random() > 0.1:  # 10%触发概率
            return players, "无事件"
        
        event_type = random.choice([
            'genetic_drift',    # 基因漂变
            'adaptive_radiation', # 适应性辐射
            'survival_crisis',  # 生存危机
            'symbiosis'         # 共生关系
        ])
        
        event_result = {
            'type': event_type,
            'week': week,
            'affected_players': [],
            'impact': 0
        }
        
        if event_type == 'genetic_drift':
            # 随机选择一名选手，技术分轻微变化
            player = random.choice(players)
            drift = random.uniform(-self.max_impact, self.max_impact)
            player.tech_score *= (1 + drift)
            event_result['impact'] = drift
            event_result['affected_players'] = [player.id]
            
        elif event_type == 'adaptive_radiation':
            # 随机选择一名选手，适应度增加
            player = random.choice(players)
            radiation = random.uniform(0, self.max_impact)
            player.adaptation = min(1.0, player.adaptation * (1 + radiation))
            event_result['impact'] = radiation
            event_result['affected_players'] = [player.id]
            
        elif event_type == 'survival_crisis':
            # 本周淘汰池扩大至3人（仅当选手数>5时）
            if len(players) > 5:
                event_result['impact'] = 1  # 标记为特殊事件
                # 在后续淘汰池确定逻辑中处理
                
        elif event_type == 'symbiosis':
            # 随机一组舞伴，互相加成
            pairs = self.find_dance_pairs(players)
            if pairs:
                pair = random.choice(pairs)
                bonus = random.uniform(0, self.max_impact)
                for player in pair:
                    player.tech_score = min(1.0, player.tech_score * (1 + bonus))
                event_result['impact'] = bonus
                event_result['affected_players'] = [p.id for p in pair]
        
        # 安全检查：防止优秀选手被误伤
        self.safety_check(players, event_result)
        
        self.events_log.append(event_result)
        return players, event_result
    
    def safety_check(self, players, event_result):
        """
        安全机制：保护技术优秀选手
        """
        # 如果事件导致技术前10%选手受影响
        tech_scores = [p.tech_score for p in players]
        top_10_threshold = np.percentile(tech_scores, 90)
        
        for player_id in event_result['affected_players']:
            player = next(p for p in players if p.id == player_id)
            if player.tech_score >= top_10_threshold and event_result['impact'] < 0:
                # 负面对优秀选手的影响减半
                if event_result['type'] == 'genetic_drift':
                    player.tech_score /= (1 + event_result['impact']/2)
                    event_result['impact'] /= 2
                    event_result['safety_override'] = True
